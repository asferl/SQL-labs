
SELECT cities.name, regions.name
FROM cities
JOIN regions ON cities.region = regions.uuid
WHERE cities.population > 350000;


SELECT cities.name
FROM cities
JOIN regions ON cities.region = regions.uuid
WHERE regions.name = 'Nord';


CREATE TABLE stations (
  id int(10) unsigned NOT NULL,
  name varchar(255) COLLATE utf8_unicode_ci NOT NULL,
  location varchar(255) COLLATE utf8_unicode_ci NOT NULL,
  ...
  PRIMARY KEY (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_unicode_ci;


CREATE TABLE `metro_map` (
  `id` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `city` varchar(255) NOT NULL,
  `line_number` int(10) unsigned NOT NULL,
  `line_name` varchar(255) NOT NULL,
  `station_name` varchar(255) NOT NULL,
  `station_order` int(10) unsigned NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
